var m,n,r,q;

procedure gcd;
  begin
    while r#0 do
      begin
        q:= m/n;
        r:= m - q * n;
        m:= n;
        n:= r;
        end
  end;
    
begin   
    m:= 36;
    n:= 24;
    if m<n then
        begin   
            r:= m;
            m:= n;
            n:= r;
        end;
    begin   
        r := 1;
        call gcd;
        write(m);
    end;
end.
